import { NextResponse } from 'next/server';
import { SearchHandler } from '@/lib/mcp/handlers/search-handler';

export async function POST(request) {
    try {
        const { query, webSearchRequested = false } = await request.json();
        
        console.log('ğŸ§ª SearchHandler test API called:', {
            query,
            webSearchRequested
        });

        const result = await SearchHandler.handleUserQuery(
            query,
            'test-user-123',
            'dark',
            webSearchRequested
        );

        console.log('ğŸ§ª SearchHandler test result:', {
            hasResponse: !!result.response,
            responseLength: result.response?.length || 0,
            webSearchUsed: result.webSearchUsed,
            webSearchFailed: result.webSearchFailed,
            quotaRemaining: result.quotaRemaining
        });

        return NextResponse.json({
            success: true,
            result: result,
            debug: {
                queryTested: query,
                webSearchRequested,
                responseContainsWebHeader: result.response?.includes('ğŸŒ From the broader web:') || false,
                responseContainsCommunityHeader: result.response?.includes('ğŸ  From flokkk Community:') || false
            }
        });

    } catch (error) {
        console.error('ğŸ§ª SearchHandler test error:', error);
        return NextResponse.json({
            success: false,
            error: error.message,
            stack: error.stack
        }, { status: 500 });
    }
}
